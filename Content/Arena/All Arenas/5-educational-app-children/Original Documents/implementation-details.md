<em>"""</em><br />

<br />

init_response = requests.post(url, headers=headers, data=init_payload)<br />

<br />

<br />
session_id = init_data["data"]["session_id"]<br />
<br />
input_node_id = None<br />
message_id = None<br />
for event in init_data["data"]["events"]:<br />
if event["event"] == "input":<br />
input_node_id = event["node_id"]<br />
message_id = event["message_id"]<br />
break<br />
<br />
if not input_node_id or not message_id:<br />
raise Exception("未能找到 input 节点或 message_id")<br />
<br />

}<br />
},<br />
"message_id": message_id,<br />
"session_id": session_id<br />
})<br />

<br />

<br />
